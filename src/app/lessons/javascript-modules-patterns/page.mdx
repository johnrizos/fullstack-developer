  { name: "Keyboard", price: 80, quantity: 1 },
  { name: "Mouse", price: 35, quantity: 2 },
  { name: "Monitor", price: 220, quantity: 1 }
];

function calculateLineTotal(item) {
  return item.price * item.quantity;
}

function calculateCartTotal(items) {
  return items.reduce(function (total, item) {
    return total + calculateLineTotal(item);
  }, 0);
}

function formatCurrency(amount) {
  return "$" + amount.toFixed(2);
}

document.querySelector("#calculate").addEventListener("click", function () {
  const total = calculateCartTotal(cartItems);
  document.querySelector("#total").textContent = "Total: " + formatCurrency(total);
});`;

# Modules & Code Organization

Όταν ο κώδικας μεγαλώνει, το πρόβλημα δεν είναι μόνο “να δουλεύει”. Πρέπει να είναι κατανοητός, testable και εύκολος να αλλάξει.

Σε πραγματικό project, χωρίζουμε τον κώδικα σε μικρά κομμάτια:

- pure functions για λογική,
- UI/event code για αλληλεπίδραση,
- modules με imports/exports,
- καθαρά ονόματα που εξηγούν τι κάνει κάθε κομμάτι.

## Pure functions

Μια pure function επιστρέφει αποτέλεσμα με βάση τα inputs της και δεν αλλάζει εξωτερική κατάσταση.

```js
function add(a, b) {
  return a + b;
}
```

Αυτό είναι εύκολο να τεσταριστεί.

## Separation of concerns

Μη βάζεις όλο τον κώδικα μέσα σε ένα click handler. Κράτα τη λογική χωριστά:

```js
function calculateTotal(items) {
  return items.reduce((total, item) => total + item.price, 0);
}
```

Το click handler απλώς καλεί τη λογική και ενημερώνει το UI.

## Modules

Σε πραγματικά projects θα έχεις:

```js
export function calculateTotal(items) {}
import { calculateTotal } from "./cart";
```

Το challenge εδώ είναι single-file για να τρέχει στο browser preview, αλλά η λογική είναι γραμμένη σαν να μπορούσε να μεταφερθεί σε module.
