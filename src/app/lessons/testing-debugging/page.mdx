  price: number;
  quantity: number;
};

export function calculateTotal(items: CartItem[]): number {
  return items.reduce((total, item) => {
    return total + item.price * item.quantity;
  }, 0);
}

export function applyDiscount(total: number, discountPercent: number): number {
  if (discountPercent < 0 || discountPercent > 100) {
    throw new Error("Discount must be between 0 and 100");
  }

  return total - total * (discountPercent / 100);
}

describe("cart utilities", () => {
  it("calculates total", () => {
    expect(calculateTotal([
      { price: 10, quantity: 2 },
      { price: 5, quantity: 1 }
    ])).toBe(25);
  });

  it("throws for invalid discount", () => {
    expect(() => applyDiscount(100, 120)).toThrow();
  });
});`;

# Testing & Debugging

Testing δεν σημαίνει “γράφω tests για να γράψω tests”. Σημαίνει ότι μπορώ να αλλάζω κώδικα χωρίς να φοβάμαι ότι έσπασα βασική συμπεριφορά.

Debugging δεν σημαίνει “βάζω console.log παντού”. Σημαίνει ότι απομονώνω το πρόβλημα, φτιάχνω υπόθεση και την ελέγχω.

## Unit tests

Unit test ελέγχει μικρή μονάδα λογικής, συνήθως function.

```ts
expect(calculateTotal(items)).toBe(25);
```

Οι pure functions είναι εύκολες για unit tests γιατί έχουν input και output χωρίς side effects.

## Integration και UI tests

Για frontend apps θα δεις συχνά:

- React Testing Library για component behavior,
- Playwright ή Cypress για end-to-end user flows.

Παράδειγμα user-flow σκέψης:

> Ο χρήστης γράφει email, πατά submit, βλέπει success ή validation error.

## Debugging workflow

Καλός τρόπος debugging:

1. Αναπαράγω το bug.
2. Μικραίνω το πρόβλημα.
3. Διαβάζω error message και stack trace.
4. Ελέγχω inputs/state/network.
5. Γράφω test αν το bug είναι σημαντικό.
