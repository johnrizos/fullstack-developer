import userEvent from "@testing-library/user-event";

type LoginFormProps = {
  onSubmit: (email: string) => void;
};

function LoginForm({ onSubmit }: LoginFormProps) {
  return (
    <form onSubmit={(event) => {
      event.preventDefault();
      const form = event.currentTarget;
      const email = new FormData(form).get("email") as string;
      onSubmit(email);
    }}>
      <label htmlFor="email">Email</label>
      <input id="email" name="email" type="email" />
      <button type="submit">Login</button>
    </form>
  );
}

test("submits email", async () => {
  const user = userEvent.setup();
  const onSubmit = vi.fn();

  render(<LoginForm onSubmit={onSubmit} />);

  await user.type(screen.getByLabelText("Email"), "ada@example.com");
  await user.click(screen.getByRole("button", { name: "Login" }));

  expect(onSubmit).toHaveBeenCalledWith("ada@example.com");
});`;

# React Testing

Frontend testing πρέπει να ελέγχει συμπεριφορά χρήστη, όχι implementation details.

Καλή ερώτηση για κάθε test:

> Αν ήμουν χρήστης, τι βλέπω και τι κάνω;

## React Testing Library

Η React Testing Library σε σπρώχνει να βρίσκεις elements όπως θα τα έβρισκε ο χρήστης ή screen reader:

```ts
screen.getByLabelText("Email");
screen.getByRole("button", { name: "Login" });
```

Αυτό ενισχύει και accessibility.

## userEvent

Το `userEvent` προσομοιώνει αλληλεπιδράσεις πιο κοντά στον χρήστη από απλά fire events.

```ts
await user.type(input, "ada@example.com");
await user.click(button);
```

## Τι να τεστάρεις

Τέσταρε:

- submit flows,
- validation messages,
- loading/error states,
- conditional rendering,
- critical user journeys.

Μην τεστάρεις:

- internal state variables απευθείας,
- private implementation details,
- styling που αλλάζει συχνά χωρίς business σημασία.
