export const tasksJs = `const tasks = [
  { title: "Μάθε HTML", done: true },
  { title: "Μάθε CSS", done: true },
  { title: "Μάθε JavaScript", done: false }
];

function renderTasks() {
  const list = document.querySelector("#tasks");

  list.innerHTML = tasks
    .map((task) => \`<li>\${task.done ? "✅" : "⬜"} \${task.title}</li>\`)
    .join("");
}

renderTasks();`;

# JavaScript Arrays & Objects

Σχεδόν κάθε πραγματική εφαρμογή δουλεύει με λίστες δεδομένων: προϊόντα, χρήστες, tasks, μηνύματα.

Ένα **array** κρατά πολλά items:

```js
const names = ["Ada", "Grace", "Linus"];
```

Ένα **object** κρατά ιδιότητες για ένα πράγμα:

```js
const task = {
  title: "Μάθε HTML",
  done: true
};
```
